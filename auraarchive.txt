let decalMesh;

function applyLogo(textureURL) {
  const texture = new THREE.TextureLoader().load(textureURL);

  const decalMaterial = new THREE.MeshStandardMaterial({
    map: texture,
    transparent: true,
    depthTest: true,
    depthWrite: false,
    polygonOffset: true,
    polygonOffsetFactor: -4,
  });

  const decalGeometry = new THREE.DecalGeometry(
    hoodie.children[0],
    new THREE.Vector3(0, 0.2, 0.6), // chest position
    new THREE.Euler(0, 0, 0),
    new THREE.Vector3(0.4, 0.4, 0.4)
  );

  if (decalMesh) scene.remove(decalMesh);
  decalMesh = new THREE.Mesh(decalGeometry, decalMaterial);
  scene.add(decalMesh);
}

function moveLogo(x, y, z) {
  if (!decalMesh) return;
  decalMesh.position.set(x, y, z);
}

function scaleLogo(s) {
  if (!decalMesh) return;
  decalMesh.scale.setScalar(s);
}
